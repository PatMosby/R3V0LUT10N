114  private List<Constraint> toConstraints(Map<String, String> filters, Class<Element> clazz) {
115  List<Constraint> constraints = new LinkedList<Constraint>();
116  if (filters != null) {
117  HashMap<String, Field> fields = Reflection.getTransitiveFields(new HashMap<String, Field>(), clazz);
118  for (Map.Entry<String, String> entry : filters.entrySet()) {
119  final String key = entry.getKey();
120  // key is the field name of an element as used in the facelet code (*.xhtml)
121  // filters.get(key) is the value that was entered by the user into the filter field
122  try {
123  // we are using reflection here to determine whether the field is a string;
124  // if so, we will use a like-constraint, otherwise an equal-constraint
125  Field field = fields.get(key);
126  if (field == null) {
127  logger.error("No such field '" + key + "'");
128  } else {
129  if (field.getType().isAssignableFrom(String.class)) {
130  // field is a string
131  constraints.add(new LikeConstraint(key, "%" + entry.getValue() + "%", AttributeType.STRING));
132  } else {
133  // FIXME: this attribute is not necessarily an integer, yet, all that matters
134  // is the fact that it does not need to be quoted.
135  constraints.add(new EqualConstraint(key, entry.getValue(), AttributeType.INTEGER));
136  }
137  }
138  } catch (SecurityException e) {
139  logger.error("Must not access field '" + key + "' " + e.getLocalizedMessage());
140  }
141  }
142  }
143  return constraints;
144  }

149  * Required for lazy loading.

151  * with the following parameters:
152  * - first: Offset of first data to start from
153  * - pageSize: Number of data to load
154  * - sortField: Name of sort field (e.g. "model" for sortBy="#{....model}")
155  * - sortOrder: SortOrder enum.
156  * - filter: Filter map with field name as key (e.g. "model" for filterBy="#{....model}") and value.
157  *
158  * Note: there is a variant load operation with the signature:
159  * List<Element> load(int first, int pageSize, String sortField, SortOrder sortOrder, Map<String,String> filters)
160  * That load operation is required in case of single ordering. The data table, however, offers
161  * multi-ordering (several fields can be used for the ordering at once).
162  */

174  return null;

184  private List<OrderBy> toOrdering(List<SortMeta> multiSortMeta) {
185  if (multiSortMeta != null) {
186  List<OrderBy> result = new ArrayList<OrderBy>();
187  for (SortMeta sm : multiSortMeta) {
188  final SortOrder sortOrder = sm.getSortOrder();
189  switch (sortOrder) {
190  case ASCENDING: result.add(new OrderBy(sm.getSortField(), true)); break;
191  case DESCENDING: result.add(new OrderBy(sm.getSortField(), false)); break;
192  case UNSORTED: /* nothing to be done */ break;
193  }
194  }
195  return result;
196  } else {
197  return null;
198  }
199  }
