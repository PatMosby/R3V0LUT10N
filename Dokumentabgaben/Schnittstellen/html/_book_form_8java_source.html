448  private synchronized List<Book> find(String identifier) {
449  if (identifier == null || identifier.isEmpty()) {
450  return new ArrayList<Book>();
451  } else {
452  // TODO: validate isbn/issn
453  if (searcher == null) {
454  // we did not get a connection initially; let's try once more now
455  try {
456  searcher = new ISBNGoogleSearch();
457  } catch (Exception e) {
458  googleNotAvailable(e);
459  return new ArrayList<Book>();
460  }
461  }
462  List<Book> result = new ArrayList<Book>();
463  try {
464  result = searcher.searchByIndustrialIdentifier(identifier);
465  int i = 1;
466  for (Book b : result) {
467  b.setId(i);
468  i++;
469  }
470  return result;
471  } catch (Exception e) {
472  logger.error("general error " + e.getMessage()
473  + " (" + e.getClass().getCanonicalName() + ")");
474  FacesContext.getCurrentInstance().addMessage(null,
475  new FacesMessage(FacesMessage.SEVERITY_WARN,
476  Messages.get("googleNotAvailable"),
477  e.getLocalizedMessage()));
478  return result;
479  }
480  }
481  }
