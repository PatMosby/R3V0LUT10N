95  private void printBookTag(PDDocument document, PDPage page, int x, int y, BookTagContent content) throws IOException {

97  final int barCodeX = x + 25;

99 
100  // One cannot add images if a content stream is still open.

102  PDXObjectImage barCodeImage = barcodeImage(document, tagCode, barCodeWidth, barCodeHeight, false);
103 
104  // Start a new content stream which will "hold" the to be created content
105  PDPageContentStream contentStream = new PDPageContentStream(document, page, true, false);
106 
107  if(barCodeImage != null) {
108  contentStream.drawImage(barCodeImage, barCodeX, barCodeY);
109  }
110  try {
111  drawCuttingAide(x, y, width, height, contentStream);
112  printText(content.firstLine, x + textLeftMargin, y + height - fontSize - textUpperMargin, normalFont, fontSize, 0, width, contentStream, false);
113  printText(content.secondLine, x + textLeftMargin, y + height - 2 * (fontSize + textUpperMargin), normalFont, fontSize, 0, width, contentStream, false);
114  printText(tagCode, barCodeX, barCodeY - fontSize - 2, normalFont, fontSize, 0, barCodeWidth, contentStream, false);
