220  private void readHeader() throws CorruptInput {
221  try {
222  String line = reader.readLine();
223  if (line != null && !line.isEmpty()) {
224  headers = splitLine(line);
225  if (headers.length == 0) {
226  throw new CorruptInput("input has empty header");
227  }
228  index = new HashMap<String, Integer>();
229  for (int i = 0; i < headers.length; i++) {
230  final String value = headers[i];
231  headers[i] = value;
232  if (value.isEmpty()) {
233  throw new CorruptInput("header column " + i + " is empty");
234  }
235  if (index.containsKey(value)) {
236  throw new CorruptInput("heading '" + value + "' is not unique");
237  }
238  index.put(value, i);
239  }
240  } else {
241  throw new CorruptInput("input has no header");
242  }
243  } catch (IOException e) {
244  closeReader();
245  throw new CorruptInput("input has no header");
246  }

256  private void readData() throws CorruptInput {
257  dataRows = new ArrayList<String[]>();
258  try {
259  int lineNumber = 1;
260  do {
261  String line = reader.readLine();
262  if (line == null) {
263  break; // last line
264  }
265  if (line.isEmpty()) {
266  throw new CorruptInput("empty row " + lineNumber);
267  }
268  String[] data = splitLine(line);
269  if (data.length != headers.length) {
270  throw new CorruptInput("row " + lineNumber + " has "
271  + (data.length > headers.length ? "too many" : "not enough")
272  + " entries; expected: " + headers.length
273  + " actual: " + data.length
274  + " content: " + line);
275  }
276  lineNumber++;
277  dataRows.add(data);
278  }
279  while (true);
280  } catch (IOException e) {
281  throw new CorruptInput("input has corrupt data row");
282  }

303  private String[] splitLine(String line) throws CorruptInput {
304  if (quote.isEmpty()) {
305  // there is no quoting, so we can use standard splitting
306  // argument -1 causes trailing empty strings not to be discarded
307  return line.split(((Character)separator).toString(), -1);
308  } else {
309  final char Quote = quote.charAt(0);
310  // we must implement our own splitting because separator
311  // is allowed to be contained in a quoted string
312  State state = State.init;
313  // the position of the first character of the current field excluding quote
314  int pos = 0;
315  // indicates whether there is any quoted quote contained in a quoted field,
316  // in which case we need to replace the double quote by a single quote
317  boolean quote_in_quoted_column = false;
318 
319  List<String> result = new LinkedList<String>();
320 
321  /* empty line is handled already */
322  for (int i = 0; i < line.length(); i++) {
323  final char c = line.charAt(i);
324  switch (state) {
325  case init:
326  if (c == Quote) {
327  /* new quoted field starts */
328  pos = i + 1;
329  state = State.in_quoted_column;
330  } else if (c == separator) {
331  /* field finished: empty field */
332  result.add("");
333  pos = i + 1;
334  quote_in_quoted_column = false;
335  state = State.in_unquoted_column;
336  } else {
337  /* new unquoted field starts */
338  pos = i;
339  quote_in_quoted_column = false;
340  state = State.in_unquoted_column;
341  }
342  break;
343  case in_quoted_column:
344  if (c == Quote) {
345  // look ahead: if Quote is followed by Quote, the Quote is an
346  // escaped Quote that does not close the quoting; it must be
347  // replaced by a single quote instead
348  if (i + 1 < line.length() && line.charAt(i + 1) == Quote) {
349  // remember that we have seen a quoted quote; it will be
350  // replaced by a single quote later
351  quote_in_quoted_column = true;
352  // skip the next quote, too
353  i++;
354  } else {
355  /* quoted field ends */
356  result.add(extract(line, pos, i, quote_in_quoted_column));
357  state = State.quoted_column_closed;
358  }
359  } else {
360  /* remain in this state */
361  }
362  break;
363  case in_unquoted_column:
364  if (c == separator) {
365  /* new column starts */
366  result.add(extract(line, pos, i, quote_in_quoted_column));
367  pos = i + 1;
368  quote_in_quoted_column = false;
369  state = State.init;
370  } else {
371  /* remain in this state */
372  }
373  break;
374  case quoted_column_closed:
375  if (c == separator) {
376  /* a new column starts */
377  pos = i + 1;
378  quote_in_quoted_column = false;
379  state = State.new_column;
380  } else {
381  /* syntax error */
382  throw new CorruptInput("separator expected in line ("
383  + line + ") "
384  + (i + 1) + " at column " + pos
385  + ": got " + c);
386  }
387  break;
388  case new_column:
389  if (c == Quote) {
390  /* new quoted column */
391  pos = pos + 1;
392  quote_in_quoted_column = false;
393  state = State.in_quoted_column;
394  } else if (c == separator) {
395  /* new empty column */
396  result.add("");
397  pos = pos + 1;
398  quote_in_quoted_column = false;
399  /* remain in this state */
400  } else {
401  /* new unquoted column */
402  /* pos has the correct value */
403  state = State.in_unquoted_column;
404  }
405  break;
406  }
407  }
408  /* for state == State.quoted_column_closed there is nothing to do when the line ends */
409  if (state == State.in_quoted_column) {
410  throw new CorruptInput("unclosed quoted value in '" + line + "' starting at position " + (pos + 1));
411  } else if (state == State.in_unquoted_column || state == State.new_column || state == State.init) {
412  /* finish the last column */
413  result.add(extract(line, pos, line.length(), quote_in_quoted_column));
414  }
415  return result.toArray(new String[0]);
416  }
417  }

436  private String extract(String string, int from, int to, boolean containsQuote) {
437  if (containsQuote) {
438  final char Quote = quote.charAt(0);
439  StringBuilder s = new StringBuilder();
440  int pos = from;
441  while (pos < to) {
442  final char current = string.charAt(pos);
443  if (current == Quote && pos + 1 < to && string.charAt(pos + 1) == Quote) {
444  // first quote in a double quote (quoted quote);
445  // we will ignore this one and pick up the second quote in the
446  // next iteration
447  } else {
448  s.append(current);
449  }
450  pos++;
451  }
452  return s.toString();
453  } else {
454  return string.substring(from, to);
455  }
456  }
