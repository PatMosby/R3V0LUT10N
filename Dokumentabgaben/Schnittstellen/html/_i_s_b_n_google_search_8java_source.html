177  private void setDateOfPublication(Volume.VolumeInfo volumeInfo, final Book book) {
178  // we get very different formats for the publication, such as 2009, 2009-07, or

180  final String date = volumeInfo.getPublishedDate();

182  String [] formats = {"yyyy-MM-dd", "yyyy-MM", "yyyy", "yy-MM-dd", "yy-MM", "yy"};
183 
184  Date googleDate = null;

187  if (googleDate != null) {
188  break;

190  }
191  if (googleDate == null) {
192  book.setDateOfPublication(null);
193  } else {
194  book.setDateOfPublication(googleDate);
195  }
196  }
