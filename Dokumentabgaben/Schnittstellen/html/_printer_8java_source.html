236  protected PDDocument createDocument(List<Content> idcontent)
237  throws IOException {
238  // Create a document and add a page to it
239  PDDocument document = new PDDocument();
240  PDPage page = new PDPage();

242  document.addPage(page);

246  // print numberOfCardsPerColumn x numberOfCardsPerRow cards per A4 page

248  // System.out.println("IDCardPrinter.createDocument() " + columnNumber + " " + rowNumber);
249  printCard(document, page, columnNumber, rowNumber, data);
250  columnNumber++;
251  if (columnNumber % numberOfCardsPerRow() == 0) {
252  // start new row
253  rowNumber++;
254  columnNumber = 0;
255  }

258  page = new PDPage();

288  protected static BufferedImage rotate(BufferedImage image, double angle){
289  final double cos = Math.abs(Math.cos(Math.toRadians(angle)));
290  final double sin = Math.abs(Math.sin(Math.toRadians(angle)));
291 
292  final int height = image.getHeight(null);
293  final int width = image.getWidth(null);
294 
295  final int newHeight = (int) Math.floor(height * cos + width * sin);
296  final int newWidth = (int) Math.floor(width * cos + height * sin);
297 
298  BufferedImage result = new BufferedImage(newWidth, newHeight, BufferedImage.TYPE_BYTE_BINARY);
299  Graphics2D graphics = result.createGraphics();
300  graphics.translate((newWidth - width) / 2, (newHeight - height) / 2);
301  graphics.rotate(Math.toRadians(angle), width / 2.0, height / 2.0);
302  graphics.drawRenderedImage(image, null);
303  graphics.dispose();
304  return result;
305  }

356  } else {
